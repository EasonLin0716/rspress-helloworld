# code block

```tsx title="hello.tsx"
type Props = {
  name: string;
};

const Hello = ({ name }: Props) => {
  return <div>Hello {name}</div>;
};
```

```ts title="hello.ts" {2}
const add = (a: number, b: number) => a + b;
const result = add(1, 2);
console.log(result);
```

```dart title="hello.dart"
void main() {
  print('Hello, World!');
}
```
